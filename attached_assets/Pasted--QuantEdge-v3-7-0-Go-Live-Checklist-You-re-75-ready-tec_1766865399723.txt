1. **Paper Trading Module** (Build this first)
```typescript
// server/paper-trading.ts
export function simulateFill(
  idea: TradeIdea,
  assetType: string
): SimulatedFill {
  const slippage = TRADING_COSTS[assetType].slippage;
  const fillPrice = idea.direction === 'long'
    ? idea.entryPrice * (1 + slippage) // Worse fill on entry
    : idea.entryPrice * (1 - slippage);
  
  return {
    ...idea,
    simulatedFillPrice: fillPrice,
    status: 'paper',
    simulatedPnL: calculatePnLWithCosts(fillPrice, idea.exitPrice || 0, assetType)
  };
}
```
- **Why**: Legal shield. Forces users to prove profitability before risking real money.
- **Requirement**: 20 paper trades with ≥55% win rate unlocks live mode.

2. **Correlation Blocking** (Critical risk control)
```typescript
// server/risk-manager.ts
export async function checkSectorExposure(userId: string, newSymbol: string) {
  const positions = await getOpenPositions(userId);
  const newSector = await getSector(newSymbol);
  
  const sectorCount = positions.filter(p => p.sector === newSector).length;
  return sectorCount >= 3; // Max 3 positions per sector
}
```
- **Why**: Prevents "diversified" bets on 5 tech stocks that all move together.
