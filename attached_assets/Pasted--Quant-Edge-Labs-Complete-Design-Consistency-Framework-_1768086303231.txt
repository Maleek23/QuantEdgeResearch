## **4. CONTENT DENSITY MANAGEMENT: Progressive Disclosure**

### **Problem**: New insights add clutter  
**Solution**: Collapsible tiers, smart defaults

```typescript
// File: client/src/hooks/use-content-density.tsx

export function useContentDensity() {
  const [density, setDensity] = useState('standard'); // 'minimal', 'standard', 'detailed'
  
  return {
    density,
    isMinimal: density === 'minimal',
    isDetailed: density === 'detailed',
    
    // Smart defaults based on user role
    getDefaultDensity(userType) {
      if (userType === 'beginner') return 'minimal';
      if (userType === 'intermediate') return 'standard';
      if (userType === 'expert') return 'detailed';
    },
    
    // Collapse/expand all
    collapseAll: () => setDensity('minimal'),
    expandAll: () => setDensity('detailed'),
  };
}

// Implementation in Command Center
<CommandCenter>
  <DensityControl className="absolute top-4 right-4 flex space-x-1 bg-slate-800/50 rounded p-1">
    <Button 
      size="sm" 
      variant={density === 'minimal' ? 'primary' : 'ghost'}
      onClick={() => setDensity('minimal')}
    >
      Minimal
    </Button>
    <Button 
      size="sm"
      variant={density === 'standard' ? 'primary' : 'ghost'}
      onClick={() => setDensity('standard')}
    >
      Standard
    </Button>
    <Button 
      size="sm"
      variant={density === 'detailed' ? 'primary' : 'ghost'}
      onClick={() => setDensity('detailed')}

      Detailed
```

## **6. MOBILE ADAPTATION: Terminal-On-The-Go**

### **Problem**: Terminal aesthetic breaks on mobile  
**Solution**: Adaptive density, collapsible sidebar, swipe gestures

```typescript
// File: client/src/hooks/use-mobile-adaptation.tsx

export function useMobileAdaptation() {
  const { isMobile } = useBreakpoint(); // < 640px
  
  return {
    // Sidebar becomes bottom sheet
    sidebarVariant: isMobile ? 'bottom-sheet' : 'fixed',
    
    // Reduce glass layers to prevent performance issues
    glassTier: isMobile ? 'minimal' : 'full',
    
    // Collapse multi-column layouts
    gridColumns: isMobile ? 1 : 12,
    
    // Simplify animations
    animations: isMobile ? false : true,
    
    // Font sizes (mobile uses smaller mono)
    fontSize: {
      metric: isMobile ? 'text-lg' : 'text-3xl',
      body: isMobile ? 'text-xs' : 'text-sm',
    },
  };
}

// Mobile Command Center Layout
<MobileCommandCenter>
  {/* Swipeable Tabs for Engines */}

      <CompactEngineCard engine="ai" />
```

## **10. FINAL PRINCIPLE: Ruthless Minimalism**

### **The 3-Question Filter for Every Element**

Before adding **any** new element to **any** page, ask:

1. **Does this help make a trading decision?** (If no, remove)
2. **Is this data displayed in monospace?** (If no, fix)
3. **Can this be collapsed in minimal mode?** (If no, add toggle)

**Design Mantra**: *"If Bloomberg Terminal doesn't have it, we don't need it."*
