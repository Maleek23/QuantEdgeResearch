### **Tier 3: Code-Level Configuration** (Experts)
```typescript
// File: user-configs/[userId]/auto-lotto-config.js

export default {
  // Entry function - user can write custom logic
  entryCondition: (signal, marketContext, userState) => {
    // User's custom condition
    if (signal.confidenceScore < 85) return false;
    if (marketContext.vix > 30) return false;
    if (userState.dailyLoss > userConfig.maxDailyLoss) return false;
    
    // Custom filter: Only trade if sector is in favor
    const allowedSectors = userConfig.allowedSectors;
    if (!allowedSectors.includes(signal.sector)) return false;
    
    return true;
  },
  
  // Position sizing formula
  positionSize: (signal, accountValue) => {
    // Custom Kelly calculation
    const kelly = calculateKelly(signal.historicalWinRate, signal.avgWin, signal.avgLoss);
    const adjustedKelly = kelly * userConfig.kellyMultiplier;
    
    // Volatility adjustment
    const atr = signal.atr;
    const volatilityPenalty = atr > 2 ? 0.7 : 1.0;
    
    return Math.min(
      accountValue * adjustedKelly * volatilityPenalty,
      accountValue * userConfig.maxPositionPercent
    );
  },
  
  // Exit function
  exitCondition: (position, marketContext) => {
    // Trailing stop based on ADX
    if (position.adx > 30) {
      return { stop: position.price * 0.92 }; // 8% trail in trend
    } else {
      return { stop: position.price * 0.95 }; // 5% trail in range
    }
  }
};
```

```typescript
function generateSuggestions(input: string): string[] {
  if (input.startsWith('/risk set')) {
    return [
      'max_position_size <percent>',
      'max_daily_loss <percent>',
      'tier <conservative|moderate|aggressive|custom>'
    ];
  }
  
  if (input.startsWith('/bot auto_lotto set')) {
    return [
      'confidence_threshold <50-100>',
      'max_daily_trades <number>',
      'confluence_minimum <number>'
    ];
  }
  
  if (input.startsWith('/symbol')) {
    // Fetch user's watchlist symbols
    return getUserWatchlistSymbols(user.id);
  }
}
```
