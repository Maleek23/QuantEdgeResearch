### **Enhanced: Graceful Degradation**
```typescript
// server/market-movers-service.ts

async function getMovers() {
  try {
    // Try live API
    return await yahooFinance.getMovers();
  } catch (error) {
    // Fallback to last successful scan
    const cached = await db.query(`
      SELECT * FROM market_movers_cache 
      WHERE cached_at > NOW() - INTERVAL '1 hour'
      ORDER BY cached_at DESC 
      LIMIT 1
    `);
    
    return {
      data: cached.movers,
      isCached: true,
      cachedAt: cached.cached_at,
      disclaimer: "Showing last successful scan. Market closed or API limit reached."
    };
  }
}
```
