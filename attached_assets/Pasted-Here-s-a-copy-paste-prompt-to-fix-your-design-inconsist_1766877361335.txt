## **Quick Component Fixes**

### **Button Component (Standardized)**

```tsx
// components/Button.tsx
export const Button = ({ variant = "glass", children, ...props }) => {
  const variants = {
    glass: "bg-[rgba(0,212,255,0.25)] border border-glass hover:bg-[rgba(0,212,255,0.35)]",
    glassSecondary: "bg-[rgba(42,42,43,0.4)] border border-glass hover:bg-[rgba(42,42,43,0.5)]",
    glassSuccess: "bg-[rgba(48,209,88,0.2)] border border-[rgba(48,209,88,0.3)] hover:bg-[rgba(48,209,88,0.3)]",
    glassDanger: "bg-[rgba(255,69,58,0.2)] border border-[rgba(255,69,58,0.3)] hover:bg-[rgba(255,69,58,0.3)]",
  };

  return (
    <button 
      className={`${variants[variant]} rounded-lg px-4 py-2 font-medium transition-all hover:translate-y-[-1px] active:translate-y-[0] active:opacity-90`}
      {...props}
    >
      {children}
    </button>
  );
};
```
