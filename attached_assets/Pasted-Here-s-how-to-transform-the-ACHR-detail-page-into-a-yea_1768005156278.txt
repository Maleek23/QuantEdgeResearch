```typescript
// server/watchlist-history-service.ts
async function getMultiYearComparison(symbol: string) {
  return db.query(`
    SELECT 
      year,
      AVG(confidence_score) as avg_grade,
      AVG(price) as avg_price,
      COUNT(*) as days_watched
    FROM watchlist_history
    WHERE symbol = $1
    GROUP BY year
    ORDER BY year DESC
    LIMIT 3
  `);
}
```
